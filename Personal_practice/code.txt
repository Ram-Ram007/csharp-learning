using System;

class Calculator
{
    public int Add(int a, int b)
    {
        return a + b;
    }

    public int Add(int a, int b, int c)
    {
        return a + b + c;
    }
}

class Program
{
    static void Main()
    {
        Calculator cal = new Calculator();
        Console.WriteLine(cal.Add(10, 20));      // 30
        Console.WriteLine(cal.Add(10, 20, 30));  // 60
    }
}
